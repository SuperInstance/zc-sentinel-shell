```bash
#!/bin/bash
# service-guard.sh - Fleet service monitor
# P0: Don't let services die silently
# P1: Find safe restart patterns
# P2: Optimize uptime

SERVICES=(
    "keeper:8900"
    "agent-api:8901"
    "holodeck:7778"
    "seed-mcp:9438"
    "shell:8846"
)

LOG_DIR="/var/log/fleet"
mkdir -p "$LOG_DIR"

check_service() {
    local service=$1
    local host=$(echo $service | cut -d: -f1)
    local port=$(echo $service | cut -d: -f2)
    
    timeout 5 curl -f "http://$host:$port/health" > /dev/null 2>&1
    local status=$?
    
    if [ $status -eq 0 ]; then
        echo "✅ $service UP"
        return 0
    elif [ $status -eq 124 ]; then
        echo "⏱️  $service TIMEOUT"
        return 1
    else
        echo "❌ $service DOWN"
        return 2
    fi
}

restart_service() {
    local service=$1
    echo "$(date) - Restarting $service" >> "$LOG_DIR/service-guard.log"
    
    case $service in
        keeper*)
            docker-compose -f /opt/fleet/docker-compose.yml restart keeper
            ;;
        agent-api*)
            systemctl restart agent-api
            ;;
        holodeck*)
            cd /opt/holodeck && cargo run --release &
            ;;
        seed-mcp*)
            pkill -f "seed-mcp" && cd /opt/seed-mcp && npm start &
            ;;
        shell*)
            /opt/shell/restart.sh
            ;;
    esac
}

# Main monitoring loop
while true; do
    for service in "${SERVICES[@]}"; do
        check_service "$service"
        if [ $? -gt 0 ]; then
            echo "ALERT: $service failed health check"
            restart_service "$service"
            sleep 10
            check_service "$service"
            if [ $? -gt 0 ]; then
                echo "CRITICAL: $service still down after restart"
                # TODO: Send alert to fleet channel
            fi
        fi
    done
    sleep 60
done
```

3. **Immediate Restart** - Auto-restart on failure with 10s verification window

- No disk/memory monitoring
- No alert escalation (only logs)
- No historical uptime tracking
- Single-point monitoring (no redundancy)
