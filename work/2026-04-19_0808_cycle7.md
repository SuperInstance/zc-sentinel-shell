# service-guard.sh - Fleet service monitor
# Watches 5 critical services, restarts if down, logs to /var/log/fleet/

SERVICES=(
    "keeper:8900"
    "agent-api:8901"
    "holodeck:7778"
    "seed-mcp:9438"
    "shell:8846"
)

LOG_DIR="/var/log/fleet"
mkdir -p $LOG_DIR

for service in "${SERVICES[@]}"; do
    name=$(echo $service | cut -d: -f1)
    port=$(echo $service | cut -d: -f2)
    
    # Check if port is listening
    if ! nc -z localhost $port > /dev/null 2>&1; then
        echo "$(date) - $name on port $port is DOWN" >> $LOG_DIR/service-guard.log
        # Restart logic would go here (systemctl restart $name)

done
